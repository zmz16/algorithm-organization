# 邻接表和邻接矩阵

## 邻接矩阵

```cpp
// 构建邻接矩阵
vector<vector<int>> graph(n + 1, vector<int>(n + 1, 0));   // 邻接矩阵，节点编号为1-n，这样可以对应起来
for (int i = 1; i <= m; i++) {
	int s, e;    // start, end
	cin >> s >> e;
	graph[s][e] = 1;
}

//打印邻接矩阵
for (int i = 1; i <= n; i++) {
	for (int j = 1; j <= n; j++) {
		cout << graph[i][j] << " ";
	}
	cout << endl;
}
```

## 邻接表

```cpp
// 构建邻接表
vector<list<int>> graph1(n + 1);    // 表示节点编号
for (int i = 1; i <= n; i++) {
	int s, e;    // start, end
	cin >> s >> e;
	graph1[s].push_back(e);
}

// 打印
for (int i = 1; i < n; i++) {
	cout << i << "->";
	for (int j : graph1[i]) {
		cout << j << " ";
	}
	cout << endl;
}
```
