# 二叉树的前中后序遍历

**递归遍历**

**深度优先搜索**

## 二叉树的定义

```cpp
/**
 * Definition for a binary tree node.
 */
struct TreeNode {
    int val;                 // 节点存储的整数值
    TreeNode *left;          // 指向左子节点的指针
    TreeNode *right;         // 指向右子节点的指针

    // 构造函数1：初始化空节点（值=0，左右子节点=nullptr）
    TreeNode() : val(0), left(nullptr), right(nullptr) {}

    // 构造函数2：初始化带值的节点（左右子节点=nullptr）
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}

    // 构造函数3：完整初始化节点（值+左右子节点）
    TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
};
```

## 前序遍历

**中左右**

```cpp
class Solution {
public:
    void traversal (TreeNode * curr, vector<int> & vec) {
        if (curr == NULL) {
            return;
        }
        vec.push_back(curr->val);   // 中
        traversal(curr->left, vec);   // 佐
        traversal(curr->right, vec);   // 右
    }

    vector<int> preorderTraversal (TreeNode * root) {
        vector<int> result;
        traversal(root, result);
        return result;
    }
};
```

## 中序遍历

**左中右**

```cpp
class Solution {
public:
    void traversal (TreeNode * curr, vector<int> & vec) {
        if (curr == nullptr) {
            return;
        }
        
        traversal(curr->left, vec);   // 左
        vec.push_back(curr->val);  // 中
        traversal(curr->right, vec);   // 右
    }

    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> result;
        traversal(root, result);
        return result;
    }
};
```

## 后序遍历

**左右中**

```cpp
class Solution {
public:
    void traversal (TreeNode * curr, vector<int> & vec) {
        if (curr == nullptr) {
            return;
        }
        
        traversal(curr->left, vec);   // 左
        traversal(curr->right, vec);   // 右
        vec.push_back(curr->val);  // 中
    }

    vector<int> postorderTraversal(TreeNode* root) {
        vector<int> result;
        traversal(root, result);
        return result;
    }
};
```
