# 二叉树的层序遍历

**广度优先搜索**算法

## 二叉树的定义

```cpp
/**
 * Definition for a binary tree node.
 */
struct TreeNode {
    int val;                 // 节点存储的整数值
    TreeNode *left;          // 指向左子节点的指针
    TreeNode *right;         // 指向右子节点的指针

    // 构造函数1：初始化空节点（值=0，左右子节点=nullptr）
    TreeNode() : val(0), left(nullptr), right(nullptr) {}

    // 构造函数2：初始化带值的节点（左右子节点=nullptr）
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}

    // 构造函数3：完整初始化节点（值+左右子节点）
    TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
};
```

## 非递归

用**二维数组**保存遍历结果

```cpp
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> result;   // 保存结果
        queue<TreeNode*> que;

        if (root) {
            que.push(root);
        }

        while ( ! que.empty()) {
            int size = que.size();
            vector<int> vec;   // 存储当前层的节点数据
            while (size--) {
                TreeNode * node = que.front();   // 取出队首元素
                que.pop();
                vec.push_back(node->val);

                if (node->left) {
                    que.push(node->left);
                }
                if (node->right) {
                    que.push(node->right);
                }
            }
            result.push_back(vec);
        }
        return result;
    }
};
```

## 递归

```cpp
class Solution {
public:
    void order(TreeNode * curr, vector<vector<int>> & result, int depth) {
        if (curr == nullptr) {   // 递归地终止条件
            return;
        }

        if (result.size() == depth) {   // 如果当前层尚未初始化，先初始化
            result.push_back(vector<int>());
        }
        result[depth].push_back(curr->val);
        order(curr->left, result, depth + 1);
        order(curr->right, result, depth + 1);
    }

    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> result;   // 保存结果
        int depth = 0;
        order(root, result, depth);
        return result;
    }
};
```
